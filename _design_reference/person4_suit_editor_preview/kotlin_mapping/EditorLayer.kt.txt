package com.suitup.editor

import androidx.compose.ui.graphics.Color
import androidx.compose.ui.graphics.BlendMode
import androidx.compose.ui.graphics.ColorFilter as ComposeColorFilter

data class EditorLayer(
    val id: String,
    val name: String,
    val angle: PreviewAngle,
    val zOrder: Int,
    val assetResId: Int,            // ex: R.drawable.layer_front_jacket_base
    val assetPath: String,          // ex: "suit_layers/front/jacket_base.png"
    val canChangeColor: Boolean,
    val canChangeFabric: Boolean,
    val visible: Boolean = true,
    val toggleable: Boolean = true
)

// Composable que renderiza todas as layers de um ângulo
// Ordenado por zOrder, com ColorFilter aplicado
@Composable
fun SuitLayerStack(
    layers: List<EditorLayer>,
    angle: PreviewAngle,
    color: Color,
    modifier: Modifier = Modifier
) {
    Box(modifier = modifier) {
        layers
            .filter { it.angle == angle && it.visible }
            .sortedBy { it.zOrder }
            .forEach { layer ->
                Image(
                    painter = painterResource(layer.assetResId),
                    contentDescription = layer.name,
                    colorFilter = if (layer.canChangeColor) {
                        ComposeColorFilter.tint(color, blendMode = BlendMode.SrcIn)
                    } else null,
                    modifier = Modifier.fillMaxSize()
                )
            }
    }
}
